<script>
    const toggleBtn = document.getElementById('dropdownToggle');
    const menu = document.getElementById('dropdownMenu');

    toggleBtn.addEventListener('click', function(e) {
        e.stopPropagation();
        menu.classList.toggle('show');
    });

    document.addEventListener('click', function(event) {
        if (!toggleBtn.contains(event.target) && !menu.contains(event.target)) {
            menu.classList.remove('show');
        }
    });

    // Updated function to handle different copy links
    function copyAndOpen(type) {
        let link = "";
        
        if (type === 'roblox') {
            link = "https://www.roblox.com/catalog/12349077059/Gaming-Shirt";
        } else if (type === 'discord') {
            link = "https://discord.gg/AGfKty2jsq";
        } else if (type === 'youtube') {
            link = "https://m.youtube.com/@BananaBalintYT";
        }
        
        // Copy to clipboard
        navigator.clipboard.writeText(link).then(() => {
            // Open in new tab
            window.open(link, '_blank');
        }).catch(() => {
            // Fallback if clipboard fails
            prompt("Copy this link manually:", link);
            window.open(link, '_blank');
        });
    }
</script>
